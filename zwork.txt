                ` : ''}

        // Safety check to prevent the 'textContent' of null error

        if (isAI && p) {

            copyBtn.onclick = () => {

              setTimeout(() => icon.textContent = 'content_copy', 2000);
            };
          }
        }

        if (isAI && p && !isHistory) {
          let i = 0;
          const typeWriter = () => {
            if (!isGenerating) return; // Exit loop if stopped
            if (i < text.length) {
              p.textContent += text.charAt(i);
              i++;
              chatFeed.scrollIntoView({ behavior: 'auto', block: 'end' });

              toggleInputState(false);
            }
          };
          typeWriter();
        } else {

          if (isAI) toggleInputState(false);
          chatFeed.scrollIntoView({ behavior: 'smooth', block: 'end' });
        }
      };

      const handleSend = async (text) => {

          .replace(/^- (.+)$/gm, '<li class="ml-4 list-disc">$1</li>')
          .replace(/^\d+\. (.+)$/gm, '<li class="ml-4 list-decimal">$1</li>')

    (() => {
      if (!window.matchMedia("(pointer: fine)").matches) return;

      const ring = document.querySelector(".cursor-ring");
      const orb = document.querySelector(".cursor-orb");
      if (!ring || !orb) return;

      const state = {
        mouseX: window.innerWidth / 2,
        mouseY: window.innerHeight / 2,
        ringX: window.innerWidth / 2,
        ringY: window.innerHeight / 2,
        orbX: window.innerWidth / 2,
        orbY: window.innerHeight / 2
      };

      const interactiveSelector = "a, button, input, label, article, [role='button']";

      const animate = () => {
        state.ringX += (state.mouseX - state.ringX) * 0.12;
        state.ringY += (state.mouseY - state.ringY) * 0.12;
        state.orbX += (state.mouseX - state.orbX) * 0.22;
        state.orbY += (state.mouseY - state.orbY) * 0.22;

        ring.style.transform = `translate3d(${state.ringX}px, ${state.ringY}px, 0) translate(-50%, -50%)`;
        orb.style.transform = `translate3d(${state.orbX}px, ${state.orbY}px, 0) translate(-50%, -50%)`;
        requestAnimationFrame(animate);
      };

      window.addEventListener("mousemove", (event) => {
        state.mouseX = event.clientX;
        state.mouseY = event.clientY;
        document.body.classList.add("cursor-active");
        document.body.classList.toggle("cursor-hover", !!event.target.closest(interactiveSelector));
      });

      window.addEventListener("mouseout", () => {
        document.body.classList.remove("cursor-active", "cursor-hover");
      });

      window.addEventListener("mousedown", () => {
        document.body.classList.add("cursor-hover");
      });

      window.addEventListener("mouseup", () => {
        document.body.classList.remove("cursor-hover");
      });

      requestAnimationFrame(animate);
    })();

    document.addEventListener('DOMContentLoaded', () => {
      const pageLoader = document.getElementById('page-loader');
      if (pageLoader) {
        const hideLoader = () => {

          window.setTimeout(() => pageLoader.remove(), 600);
        };

        window.addEventListener('load', () => {

        window.setTimeout(hideLoader, 2600);

      // 1. Interactive Mouse Tilt & Shine Effect
      const mouseTiltElements = document.querySelectorAll('.mouse-tilt');

      mouseTiltElements.forEach(el => {
        el.addEventListener('mousemove', e => {
          const rect = el.getBoundingClientRect();
          const x = e.clientX - rect.left;
          const y = e.clientY - rect.top;

          const centerX = rect.width / 2;
          const centerY = rect.height / 2;

          const rotateX = ((y - centerY) / centerY) * -8;
          const rotateY = ((x - centerX) / centerX) * 8;

          el.style.setProperty('--mouse-x', `${(x / rect.width) * 100}%`);
          el.style.setProperty('--mouse-y', `${(y / rect.height) * 100}%`);
          el.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale(1.02)`;
        });

        el.addEventListener('mouseleave', () => {
          el.style.transform = `rotateX(0deg) rotateY(0deg) scale(1)`;
        });
      });

      // 2. Focus Particles System
      const canvas = document.getElementById('focus-particles');
      const ctx = canvas.getContext('2d');
      const section = document.getElementById('environment-section');
      let particles = [];
      let mouse = { x: null, y: null };

      const resizeCanvas = () => {
        canvas.width = section.offsetWidth;
        canvas.height = section.offsetHeight;
      };

      window.addEventListener('resize', resizeCanvas);
      resizeCanvas();

      section.addEventListener('mousemove', (e) => {
        const rect = section.getBoundingClientRect();
        mouse.x = e.clientX - rect.left;
        mouse.y = e.clientY - rect.top;
      });

      section.addEventListener('mouseleave', () => {
        mouse.x = null;
        mouse.y = null;
      });

      class Particle {
        constructor() {
          this.init();
        }
        init() {
          this.x = Math.random() * canvas.width;
          this.y = Math.random() * canvas.height;
          this.size = Math.random() * 2 + 1;
          this.speedX = (Math.random() - 0.5) * 0.4;
          this.speedY = (Math.random() - 0.5) * 0.4;
          this.opacity = Math.random() * 0.5 + 0.1;
        }
        update() {
          this.x += this.speedX;
          this.y += this.speedY;

          if (mouse.x && mouse.y) {
            const dx = mouse.x - this.x;
            const dy = mouse.y - this.y;
            const distance = Math.sqrt(dx * dx + dy * dy);
            if (distance < 150) {
              this.x -= dx * 0.01;
              this.y -= dy * 0.01;
            }
          }

          if (this.x < 0 || this.x > canvas.width || this.y < 0 || this.y > canvas.height) {
            this.init();
          }
        }
        draw() {
          ctx.fillStyle = `rgba(111, 251, 190, ${this.opacity})`;
          ctx.beginPath();
          ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
          ctx.fill();
        }
      }

      const initParticles = () => {
        particles = [];
        for (let i = 0; i < 60; i++) {
          particles.push(new Particle());
        }
      };
      initParticles();

      const animateParticles = () => {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        particles.forEach(p => {
          p.update();
          p.draw();
        });
        requestAnimationFrame(animateParticles);
      };
      animateParticles();

      // 3. Existing Scroll Reveal Logic
      const revealOptions = {

        rootMargin: "0px 0px -100px 0px"
      };

      const revealObserver = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            entry.target.classList.add('visible');
          }
        });
      }, revealOptions);

      document.querySelectorAll('.reveal-fade-up, .reveal-scale-in').forEach((el) => {
        revealObserver.observe(el);
      });

      // 4. Navigation glass effect on scroll

      const navLogo = document.getElementById('nav-logo');
      const navLinks = nav.querySelectorAll('.nav-link-hover');

      window.addEventListener('scroll', () => {

          nav.classList.remove('py-4', 'py-6', 'sm:py-6');
          navLogo.classList.add('text-primary');
          navLogo.classList.remove('text-white');
          navLinks.forEach(link => {
            if (!link.classList.contains('text-primary')) {
              link.classList.add('text-on-surface-variant');
              link.classList.remove('text-white/80');
            }
          });
        } else {
          nav.classList.remove('glass-nav', 'py-3', 'sm:py-4', 'shadow-2xl', 'shadow-primary/5');
          nav.classList.add('py-4', 'sm:py-6');
          navLogo.classList.remove('text-primary');
          navLogo.classList.add('text-white');
          navLinks.forEach(link => {
            if (!link.classList.contains('text-primary')) {
              link.classList.remove('text-on-surface-variant');
              link.classList.add('text-white/80');
            }
          });
        }
      });
    });

    function startCooldown(seconds = 60) {
      clearInterval(cooldownInterval);
      resendCooldown = seconds;
      const resendBtn = document.getElementById('resendOtpBtn');
      const timerEl = document.getElementById('cooldownTimer');
      
      resendBtn.disabled = true;
      
      cooldownInterval = setInterval(() => {
        if (resendCooldown <= 0) {
          clearInterval(cooldownInterval);
          resendBtn.disabled = false;
          timerEl.textContent = "You can now request a new code.";
        } else {
          resendBtn.disabled = true;
          timerEl.textContent = `Resend available in ${resendCooldown}s`;
          resendCooldown--;
        }
      }, 1000);
    }

    async function handleInitialSubmit(event) {
      event.preventDefault();
      
      const email = document.getElementById('email').value.trim();
      
      if (!isOtpSentState) {
        // Step 1: Request OTP
        const submitBtn = document.getElementById('submitBtn');
        submitBtn.disabled = true;
        submitBtn.innerHTML = `<span>Sending...</span><span class="material-symbols-outlined text-[18px] animate-spin">sync</span>`;
        
        try {
          const res = await fetch('/api/auth/request-otp', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'include',
            body: JSON.stringify({ email })
          });
          const data = await res.json();
          
          if (!res.ok) throw new Error(data.error || 'Failed to request OTP');
          
          // Switch to OTP entering view
          isOtpSentState = true;
          document.getElementById('formFields').classList.add('hidden');
          document.getElementById('otpSection').classList.remove('hidden');
          document.getElementById('otp').setAttribute('required', 'true');
          
          submitBtn.disabled = false;
          submitBtn.innerHTML = `<span>Verify & Login</span><span class="material-symbols-outlined text-[18px]">vpn_key</span>`;
          
          showCustomModal('OTP Sent!', 'Please check your email address for the 6-digit verification code.', 'success');
          startCooldown(60);
        } catch (err) {
          console.error(err);
          showCustomModal('Login Failed', err.message);
          submitBtn.disabled = false;
          submitBtn.innerHTML = `<span>Send OTP</span><span class="material-symbols-outlined text-[18px]">arrow_forward</span>`;
        }
      } else {
        // Step 2: Verify OTP
        const otp = document.getElementById('otp').value.trim();
        if (otp.length !== 6) {
          showCustomModal('Invalid Code', 'Please enter a 6-digit verification code.');
          return;
        }
        
        const submitBtn = document.getElementById('submitBtn');
        submitBtn.disabled = true;
        submitBtn.innerHTML = `<span>Verifying...</span><span class="material-symbols-outlined text-[18px] animate-spin">sync</span>`;
        
        try {
          const verifyRes = await fetch('/api/auth/verify-otp', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'include',
            body: JSON.stringify({ email, otp })
          });
          const verifyData = await verifyRes.json();
          
          if (!verifyRes.ok) throw new Error(verifyData.error || 'Verification failed');
          
          // Redirect to main index
          showCustomModal('Logged In!', 'Welcome back to Gradee.', 'success');
          setTimeout(() => {
            window.location.href = 'rooms.html';
          }, 1500);
        } catch (err) {
          console.error(err);
          showCustomModal('Verification Failed', err.message);
          submitBtn.disabled = false;
          submitBtn.innerHTML = `<span>Verify & Login</span><span class="material-symbols-outlined text-[18px]">vpn_key</span>`;
        }
      }
    }

    async function handleResendOtp() {
      const email = document.getElementById('email').value.trim();
      const resendBtn = document.getElementById('resendOtpBtn');
      
      resendBtn.disabled = true;
      
      try {
        const res = await fetch('/api/auth/request-otp', {

          credentials: 'include',
          body: JSON.stringify({ email })
        });

        showCustomModal('New Code Sent!', 'A fresh 6-digit verification code has been dispatched to your email.', 'success');
        startCooldown(60);
      } catch (err) {
        console.error(err);
        showCustomModal('Resend Failed', err.message);
        resendBtn.disabled = false;
      }
    }

  <script>
    document.addEventListener("DOMContentLoaded", async () => {
      const contactsPanel = document.getElementById("contactsPanel");
      const contactsList = document.getElementById("contactsList");
      const messagesArea = document.getElementById("messagesArea");
      const chatHeader = document.getElementById("chatHeader");
      const chatForm = document.getElementById("chatForm");
      const chatInput = document.getElementById("chatInput");
      const cursor = document.getElementById("cursor");
      const activeAvatar = document.getElementById("activeAvatar");
      const activeName = document.getElementById("activeName");
      const activeRole = document.getElementById("activeRole");
      const closeChat = document.getElementById("closeChat");
      const clearChatBtn = document.getElementById("clearChatBtn");
      let threadEl = null;

      const rtcConfig = {
        iceServers: [
          { urls: "stun:stun.l.google.com:19302" },
          { urls: "stun:stun1.l.google.com:19302" }
        ]
      };

      let authUser = null;
      let socket = null;
      let activeChatUserId = null;
      let socketPresence = new Map();
      const users = new Map();
      const messageHistory = new Map();
      const peerState = new Map();
      const unreadCounts = new Map();

      function getStorageKey() {
        return authUser ? `gradee_message_history_${authUser.id}` : null;
      }

      function saveMessageHistory() {
        const key = getStorageKey();
        if (!key) return;

        const payload = {};
        for (const [userId, history] of messageHistory.entries()) {
          payload[userId] = {
            history,
            unreadCount: unreadCounts.get(userId) || 0,
          };
        }

        localStorage.setItem(key, JSON.stringify(payload));
      }

      function loadMessageHistory() {
        const key = getStorageKey();
        if (!key) return;

        const raw = localStorage.getItem(key);
        if (!raw) return;

        try {
          const parsed = JSON.parse(raw);
          for (const [userId, value] of Object.entries(parsed)) {
            if (Array.isArray(value)) {
              messageHistory.set(userId, value);
              unreadCounts.set(userId, 0);
            } else if (value && Array.isArray(value.history)) {
              messageHistory.set(userId, value.history);
              unreadCounts.set(userId, Number(value.unreadCount) || 0);
            }
          }
        } catch (_error) {
          localStorage.removeItem(key);
        }
      }

      function getUserAvatar(user) {
        return window.GradeeAuth.getAvatarUrl(user);
      }

      function escapeHtml(value) {
        return String(value)
          .replace(/&/g, "&amp;")
          .replace(/</g, "&lt;")
          .replace(/>/g, "&gt;")
          .replace(/"/g, "&quot;")
          .replace(/'/g, "&#39;");
      }

      function getHistory(userId) {
        if (!messageHistory.has(userId)) {
          messageHistory.set(userId, []);
        }
        return messageHistory.get(userId);
      }

      function appendHistory(userId, item) {
        const history = getHistory(userId);
        history.push(item);
        saveMessageHistory();
      }

      function clearHistory(userId) {
        messageHistory.set(userId, []);
        unreadCounts.set(userId, 0);
        saveMessageHistory();
      }

      function getUnreadCount(userId) {
        return unreadCounts.get(userId) || 0;
      }

      function markConversationRead(userId) {
        if (!userId) return;
        unreadCounts.set(userId, 0);
        saveMessageHistory();
      }

      function incrementUnread(userId) {
        unreadCounts.set(userId, getUnreadCount(userId) + 1);
        saveMessageHistory();
      }

      function isUserOnline(userId) {
        const record = socketPresence.get(userId);
        return Boolean(record && record.primarySocketId);
      }

      function getTargetSocketId(userId) {
        return socketPresence.get(userId)?.primarySocketId || null;
      }

      function getPeerRecord(userId) {
        if (!peerState.has(userId)) {
          peerState.set(userId, {
            pc: null,
            channel: null,
            targetSocketId: null,
            makingOffer: false,
            polite: false,
            remoteDescriptionSet: false,
          });
        }
        return peerState.get(userId);
      }

      async function ensurePeerConnection(userId, targetSocketId, initiate = false) {
        const record = getPeerRecord(userId);
        record.targetSocketId = targetSocketId;
        record.polite = authUser.id > userId;

        if (!record.pc) {
          const pc = new RTCPeerConnection(rtcConfig);
          record.pc = pc;

          pc.onicecandidate = event => {
            if (!event.candidate || !record.targetSocketId) return;
            socket.emit("webrtc:ice-candidate", {
              targetSocketId: record.targetSocketId,
              candidate: event.candidate,
            });
          };

          pc.onconnectionstatechange = () => {
            if (activeChatUserId === userId) {
              updateHeader(users.get(userId));
            }

            if (["disconnected", "failed", "closed"].includes(pc.connectionState)) {
              record.channel = null;
            }
          };

          pc.ondatachannel = event => {
            attachDataChannel(userId, record, event.channel);
          };
        }

        if (initiate && !record.channel) {
          const channel = record.pc.createDataChannel("chat");
          attachDataChannel(userId, record, channel);
          await createOffer(userId);
        }

        return record;
      }

      function attachDataChannel(userId, record, channel) {
        record.channel = channel;

        channel.onopen = () => {
          if (activeChatUserId === userId) {
            updateHeader(users.get(userId));
          }
        };

        channel.onclose = () => {
          if (activeChatUserId === userId) {
            updateHeader(users.get(userId));
          }
        };

        channel.onmessage = event => {
          const text = String(event.data || "");
          appendHistory(userId, {
            type: "received",
            text,
            at: Date.now(),
          });

          const user = users.get(userId);
          if (activeChatUserId === userId) {
            markConversationRead(userId);
            appendMessageToUI("received", text, getUserAvatar(user));
          } else {
            incrementUnread(userId);
          }

          renderContacts();
        };
      }

      async function createOffer(userId) {
        const record = getPeerRecord(userId);
        if (!record.pc || !record.targetSocketId) return;

        try {
          record.makingOffer = true;
          const offer = await record.pc.createOffer();
          await record.pc.setLocalDescription(offer);
          socket.emit("webrtc:offer", {
            targetSocketId: record.targetSocketId,
            description: record.pc.localDescription,
          });
        } finally {
          record.makingOffer = false;
        }
      }

      async function handleOffer({ from, fromSocketId, description }) {
        users.set(from.id, from);
        const record = await ensurePeerConnection(from.id, fromSocketId, false);
        const pc = record.pc;
        const offerCollision =
          description.type === "offer" &&
          (record.makingOffer || pc.signalingState !== "stable");

        if (offerCollision && !record.polite) {
          return;
        }

        await pc.setRemoteDescription(description);
        const answer = await pc.createAnswer();
        await pc.setLocalDescription(answer);
        socket.emit("webrtc:answer", {
          targetSocketId: fromSocketId,
          description: pc.localDescription,
        });

        renderContacts();
        if (activeChatUserId === from.id) {
          updateHeader(from);
        }
      }

      async function handleAnswer({ from, fromSocketId, description }) {
        const record = getPeerRecord(from.id);
        record.targetSocketId = fromSocketId;
        if (!record.pc) return;
        await record.pc.setRemoteDescription(description);
        if (activeChatUserId === from.id) {
          updateHeader(from);
        }
      }

      async function handleIceCandidate({ from, fromSocketId, candidate }) {
        const record = await ensurePeerConnection(from.id, fromSocketId, false);
        if (!record.pc || !candidate) return;

        try {
          await record.pc.addIceCandidate(candidate);
        } catch (error) {
          console.error("ICE candidate error:", error);
        }
      }

      function getConnectionLabel(userId) {
        const record = getPeerRecord(userId);
        if (record.channel?.readyState === "open") {
          return "Secure peer chat connected";
        }

        if (isUserOnline(userId)) {
          return "Online • establishing peer channel";
        }

        return "Offline";
      }

      function updateHeader(user) {
        if (!user) return;
        activeAvatar.src = getUserAvatar(user);
        activeName.textContent = user.name;
        activeRole.textContent = getConnectionLabel(user.id);
        clearChatBtn.classList.remove("hidden");
      }

      function formatDayLabel(timestamp) {
        const date = new Date(timestamp);
        const today = new Date();
        const yesterday = new Date();
        yesterday.setDate(today.getDate() - 1);

        const sameDay = (a, b) =>
          a.getFullYear() === b.getFullYear() &&
          a.getMonth() === b.getMonth() &&
          a.getDate() === b.getDate();

        if (sameDay(date, today)) return "Today";
        if (sameDay(date, yesterday)) return "Yesterday";
        return date.toLocaleDateString([], { day: "numeric", month: "short", year: "numeric" });
      }

      function maybeAppendDaySeparator(timestamp, lastTimestamp) {
        const currentLabel = formatDayLabel(timestamp);
        const previousLabel = lastTimestamp ? formatDayLabel(lastTimestamp) : null;
        if (currentLabel === previousLabel) return;

        const separator = document.createElement("div");
        separator.className = "flex justify-center py-1";
        separator.innerHTML = `<span class="px-4 py-1.5 rounded-full glass text-[10px] font-bold uppercase tracking-widest text-ink/40">${currentLabel}</span>`;
        threadEl.appendChild(separator);
      }

      function createMessage(type, text, img = null, timestamp = Date.now()) {
        const isSent = type === "sent";
        return `
          <div class="flex ${isSent ? 'justify-end' : 'justify-start'} group">
            <div class="flex min-w-0 max-w-[min(88%,42rem)] items-end gap-3 ${isSent ? 'flex-row-reverse' : 'flex-row'}">
              ${!isSent ? `<img src="${img}" class="h-8 w-8 rounded-xl object-cover self-end" />` : ''}
              <div class="min-w-0 ${isSent ? 'bg-plum text-white shadow-soft rounded-t-2xl rounded-bl-2xl' : 'panel text-ink rounded-t-2xl rounded-br-2xl'} px-5 py-3 text-sm">
                <div class="break-words whitespace-pre-wrap leading-relaxed">${escapeHtml(text)}</div>
                <div class="text-[9px] mt-1 opacity-50 text-right">${new Date(timestamp).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}</div>
              </div>
            </div>
          </div>
        `;
      }

      function appendMessageToUI(type, text, avatar = null, timestamp = Date.now()) {
        const wrapper = document.createElement("div");
        wrapper.innerHTML = createMessage(type, text, avatar, timestamp);
        threadEl.appendChild(wrapper.firstElementChild);
        messagesArea.scrollTop = messagesArea.scrollHeight;
      }

      function renderConversation(userId) {
        const user = users.get(userId);
        const history = getHistory(userId);
        messagesArea.innerHTML = `
          <div class="mx-auto flex w-full max-w-4xl flex-col gap-5">
            <div class="flex justify-center">
              <span class="px-4 py-1.5 rounded-full glass text-[10px] font-bold uppercase tracking-widest text-ink/40">Peer-to-peer WebRTC chat</span>
            </div>
            <div id="messageThread" class="flex flex-col gap-4"></div>
          </div>
        `;
        threadEl = document.getElementById("messageThread");
        let lastTimestamp = null;

        for (const item of history) {
          maybeAppendDaySeparator(item.at, lastTimestamp);
          appendMessageToUI(
            item.type,
            item.text,
            item.type === "received" ? getUserAvatar(user) : null,
            item.at
          );
          lastTimestamp = item.at;
        }
      }

      function getPreviewText(userId) {
        const history = getHistory(userId);
        if (!history.length) {
          return isUserOnline(userId) ? "Online and ready to connect" : "Offline";
        }

        const last = history[history.length - 1];
        return last.type === "sent" ? `You: ${last.text}` : last.text;
      }

      function renderContacts() {
        contactsList.innerHTML = "";
        const sorted = Array.from(users.values()).sort((a, b) => a.name.localeCompare(b.name));

        for (const user of sorted) {
          const online = isUserOnline(user.id);
          const unread = getUnreadCount(user.id);
          const contact = document.createElement("button");
          contact.className = `w-full flex items-center gap-4 px-4 py-4 rounded-[1.5rem] transition-all hover:bg-white/40 text-left group ${activeChatUserId === user.id ? 'bg-white/35' : ''}`;
          contact.onclick = () => openChat(user);
          contact.innerHTML = `
            <div class="relative shrink-0">
              <img src="${getUserAvatar(user)}" class="h-12 w-12 rounded-2xl object-cover shadow-soft" />
              <span class="absolute -bottom-1 -right-1 h-3.5 w-3.5 rounded-full border-2 border-white ${online ? 'bg-mint' : 'bg-ink/20'}"></span>
            </div>
            <div class="flex-1 overflow-hidden">
              <div class="flex justify-between items-center gap-3">
                <span class="font-bold text-ink truncate">${user.name}</span>
                <div class="flex items-center gap-2 shrink-0">
                  ${unread > 0 ? `<span class="min-w-5 rounded-full bg-plum px-1.5 py-0.5 text-center text-[10px] font-bold text-white">${unread > 99 ? '99+' : unread}</span>` : ''}
                  <span class="text-[10px] ${online ? 'text-plum' : 'text-ink/40'}">${online ? 'online' : 'offline'}</span>
                </div>
              </div>
              <p class="text-xs text-ink/60 truncate mt-0.5">${getPreviewText(user.id)}</p>
            </div>
          `;
          contactsList.appendChild(contact);
        }
      }

      async function openChat(user) {
        activeChatUserId = user.id;
        if (window.innerWidth < 1024) {
          contactsPanel.classList.add("hidden");
        }

        chatHeader.classList.remove("hidden");
        chatForm.classList.remove("hidden");
        markConversationRead(user.id);
        updateHeader(user);
        renderConversation(user.id);
        renderContacts();

        const targetSocketId = getTargetSocketId(user.id);
        if (targetSocketId) {
          await ensurePeerConnection(user.id, targetSocketId, true);
          updateHeader(user);
        }
      }

      closeChat.onclick = () => {
        contactsPanel.classList.remove("hidden");
        activeChatUserId = null;
        threadEl = null;
        chatHeader.classList.add("hidden");
        chatForm.classList.add("hidden");
        clearChatBtn.classList.add("hidden");
        messagesArea.innerHTML = `<div class="flex h-full flex-col items-center justify-center text-center opacity-40"><span class="material-symbols-outlined text-6xl mb-4">forum</span><p class="font-serif text-xl">Select a scholar to start collaborating</p></div>`;
        renderContacts();
      };

      clearChatBtn.onclick = () => {
        if (!activeChatUserId) return;
        clearHistory(activeChatUserId);
        renderConversation(activeChatUserId);
        renderContacts();
      };

      chatForm.onsubmit = async e => {
        e.preventDefault();
        const text = chatInput.value.trim();
        if (!text || !activeChatUserId) return;

        const target = users.get(activeChatUserId);
        const targetSocketId = getTargetSocketId(activeChatUserId);
        if (!targetSocketId) {
          appendMessageToUI("received", "This user is offline right now. Ask them to open the Messages page.", getUserAvatar(target));
          return;
        }

        const record = await ensurePeerConnection(activeChatUserId, targetSocketId, true);
        if (!record.channel || record.channel.readyState !== "open") {
          appendMessageToUI("received", "Secure channel is still connecting. Please try again in a moment.", getUserAvatar(target));
          return;
        }

        record.channel.send(text);
        const item = { type: "sent", text, at: Date.now() };
        appendHistory(activeChatUserId, item);
        appendMessageToUI("sent", text, null, item.at);
        chatInput.value = "";
        renderContacts();
      };

      socket = io({
        withCredentials: true,
      });

      socket.on("session:ready", payload => {
        socketPresence = new Map((payload.onlineUsers || []).map(item => [item.userId, item]));
        renderContacts();
      });

      socket.on("presence:update", items => {
        socketPresence = new Map((items || []).map(item => [item.userId, item]));
        renderContacts();
        if (activeChatUserId && users.has(activeChatUserId)) {
          updateHeader(users.get(activeChatUserId));
        }
      });

      socket.on("webrtc:offer", handleOffer);
      socket.on("webrtc:answer", handleAnswer);
      socket.on("webrtc:ice-candidate", handleIceCandidate);

      authUser = await window.GradeeAuth.requireUser({ redirectTo: "login.html" });
      if (!authUser) return;
      loadMessageHistory();

      const response = await fetch("/api/auth/users", {
        credentials: "include",
      });
      const data = await response.json();
      if (!response.ok) {
        throw new Error(data.error || "Failed to load users.");
      }

      for (const user of data.users) {
        users.set(user.id, user);
      }

      renderContacts();

      window.addEventListener("mousemove", e => {
        cursor.style.transform = `translate3d(${e.clientX}px, ${e.clientY}px, 0)`;
      });

      (() => {
        const currentPath = window.location.pathname.split("/").pop() || "index.html";
        document.querySelectorAll(".nav-mobile-link").forEach(link => {
          const href = link.getAttribute("href");
          if (href && href !== "#" && currentPath.includes(href)) {
            link.classList.remove("text-ink/40");
            if (link.classList.contains("bg-plum")) {
              link.classList.add("ring-4", "ring-plum/20");
            } else {
              link.classList.add("text-plum", "bg-plum/10");
            }
          }
        });
      })();
    });

    document.addEventListener("DOMContentLoaded", async () => {
      const profileAvatar = document.getElementById("profileAvatar");
      const heroName = document.getElementById("heroName");
      const heroMeta = document.getElementById("heroMeta");
      const nameInput = document.getElementById("nameInput");
      const usernameInput = document.getElementById("usernameInput");
      const emailInput = document.getElementById("emailInput");
      const joinedAt = document.getElementById("joinedAt");
      const verificationStatus = document.getElementById("verificationStatus");
      const publicHandle = document.getElementById("publicHandle");
      const profileForm = document.getElementById("profileForm");
      const saveBtn = document.getElementById("saveBtn");
      const resetBtn = document.getElementById("resetBtn");
      const logoutBtnHeader = document.getElementById("logoutBtnHeader");
      const logoutBtnSidebar = document.getElementById("logoutBtnSidebar");

      let initialUser = null;

      function renderUser(user) {
        const avatar = window.GradeeAuth.getAvatarUrl(user);
        profileAvatar.src = avatar;
        heroName.textContent = user.name;
        heroMeta.textContent = `${user.email} • @${user.username}`;
        nameInput.value = user.name;
        usernameInput.value = user.username;
        emailInput.value = user.email;
        joinedAt.textContent = user.createdAt
          ? new Date(user.createdAt).toLocaleDateString([], { day: "numeric", month: "long", year: "numeric" })
          : "-";
        verificationStatus.textContent = user.isVerified ? "Verified account" : "Verification pending";
        publicHandle.textContent = `@${user.username}`;
      }

      const user = await window.GradeeAuth.requireUser({ redirectTo: "login.html" });
      if (!user) return;

      initialUser = user;
      renderUser(user);

      profileForm.onsubmit = async event => {
        event.preventDefault();
        saveBtn.disabled = true;
        saveBtn.textContent = "Saving...";

        try {
          const updatedUser = await window.GradeeAuth.updateProfile({
            name: nameInput.value.trim(),
            username: usernameInput.value.trim(),
          });
          initialUser = updatedUser;
          renderUser(updatedUser);
          saveBtn.textContent = "Saved";
          setTimeout(() => {
            saveBtn.textContent = "Save Changes";
            saveBtn.disabled = false;
          }, 900);
        } catch (error) {
          saveBtn.textContent = error.message || "Save failed";
          setTimeout(() => {
            saveBtn.textContent = "Save Changes";
            saveBtn.disabled = false;
          }, 1600);
        }
      };

      resetBtn.onclick = () => {
        if (!initialUser) return;
        renderUser(initialUser);
      };

      logoutBtnHeader.onclick = () => window.GradeeAuth.logout({ redirectTo: "login.html" });
      if (logoutBtnSidebar) {
        logoutBtnSidebar.onclick = () => window.GradeeAuth.logout({ redirectTo: "login.html" });
      }
    });
  </script>

      function getPeerRecord(userId, user = null) {

      function closePeer(userId) {
        const record = roomPeers.get(userId);
        if (!record) return;

        try {
          record.stream?.getTracks().forEach(track => track.stop());
          record.pc?.close();
        } catch (_error) {
        }

        removeRemoteTile(record);
        roomPeers.delete(userId);
        renderPresence();
      }

      function syncPeerSenders(record) {
        if (!record.pc) return;
        const outgoing = screenStream || localStream;
        for (const kind of ["video", "audio"]) {
          const track = outgoing?.getTracks().find(item => item.kind === kind) || null;
          const sender = record.senders[kind];
          if (sender) {
            sender.replaceTrack(track);
          } else if (track) {
            record.senders[kind] = record.pc.addTrack(track, outgoing);
          }
        }
      }

      async function ensurePeerConnection(user, initiate = false) {
        const record = getPeerRecord(user.userId || user.id, user);
        record.targetSocketId = user.primarySocketId || record.targetSocketId;
        record.polite = authUser.id > record.userId;

        if (!record.pc) {
          const pc = new RTCPeerConnection(rtcConfig);
          record.pc = pc;
          syncPeerSenders(record);
          createRemoteTile(record);

          pc.onicecandidate = event => {
            if (!event.candidate || !record.targetSocketId) return;
            socket.emit("webrtc:ice-candidate", {
              targetSocketId: record.targetSocketId,
              candidate: event.candidate
            });
          };

          pc.ontrack = event => {
            const [stream] = event.streams;
            if (!stream) return;
            record.stream = stream;
            if (!record.tile) {
              createRemoteTile(record);
            }
            const video = record.tile.querySelector("video");
            const placeholder = record.tile.querySelector("[data-empty]");
            const status = record.tile.querySelector("[data-status]");
            video.srcObject = stream;
            placeholder?.classList.add("hidden");
            status.textContent = "Live in room";
          };

          pc.onconnectionstatechange = () => {
            if (record.tile) {
              const status = record.tile.querySelector("[data-status]");
              if (status) {
                status.textContent = pc.connectionState === "connected" ? "Live in room" : `Status: ${pc.connectionState}`;
              }
            }

            if (["disconnected", "failed", "closed"].includes(pc.connectionState)) {
              if (!activeRoomId) return;
              renderPresence();
            }
          };
        } else {
          syncPeerSenders(record);
        }

        if (initiate) {
          await createOffer(record.userId);
        }

        renderPresence();
        return record;
      }

      async function createOffer(userId) {
        const record = roomPeers.get(userId);
        if (!record?.pc || !record.targetSocketId) return;

        try {
          record.makingOffer = true;
          const offer = await record.pc.createOffer();
          await record.pc.setLocalDescription(offer);
          socket.emit("webrtc:offer", {
            targetSocketId: record.targetSocketId,
            description: record.pc.localDescription
          });
        } finally {
          record.makingOffer = false;
        }
      }

      async function handleOffer({ from, fromSocketId, description }) {
        const record = await ensurePeerConnection({
          userId: from.id,
          name: from.name,
          username: from.username,
          primarySocketId: fromSocketId
        }, false);
        const pc = record.pc;
        const offerCollision =
          description.type === "offer" &&
          (record.makingOffer || pc.signalingState !== "stable");

        if (offerCollision && !record.polite) {
          return;
        }

        await pc.setRemoteDescription(description);
        syncPeerSenders(record);
        const answer = await pc.createAnswer();
        await pc.setLocalDescription(answer);
        socket.emit("webrtc:answer", {
          targetSocketId: fromSocketId,
          description: pc.localDescription
        });
        renderPresence();
      }

      async function handleAnswer({ from, fromSocketId, description }) {

          name: from.name,
          username: from.username,
          primarySocketId: fromSocketId

        record.targetSocketId = fromSocketId;
        if (!record.pc) return;
        await record.pc.setRemoteDescription(description);
        renderPresence();

      async function handleIceCandidate({ from, fromSocketId, candidate }) {

          primarySocketId: fromSocketId

      async function ensureLocalStream() {

          refreshLocalVisuals();
          return localStream;

          track.enabled = micEnabled;

        localStream.getVideoTracks().forEach(track => {
          track.enabled = cameraEnabled;

        refreshLocalVisuals();
        syncAllPeerSenders();
        return stream;
      }

    function startCooldown(seconds = 60) {
      clearInterval(cooldownInterval);

      const resendBtn = document.getElementById('resendOtpBtn');
      const timerEl = document.getElementById('cooldownTimer');
      
      resendBtn.disabled = true;
      
      cooldownInterval = setInterval(() => {
        if (resendCooldown <= 0) {
          clearInterval(cooldownInterval);
          resendBtn.disabled = false;
          timerEl.textContent = "You can now request a new code.";
        } else {
          resendBtn.disabled = true;
          timerEl.textContent = `Resend available in ${resendCooldown}s`;

    async function handleInitialSubmit(event) {
      event.preventDefault();
      
      const email = document.getElementById('email').value.trim();
      const name = document.getElementById('name').value.trim();
      const username = document.getElementById('username').value.trim();
      
      if (!isOtpSentState) {
        // Step 1: Request OTP
        const submitBtn = document.getElementById('submitBtn');
        submitBtn.disabled = true;
        submitBtn.innerHTML = `<span>Sending...</span><span class="material-symbols-outlined text-[18px] animate-spin">sync</span>`;
        
        try {
          const res = await fetch('/api/auth/request-otp', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'include',
            body: JSON.stringify({ email, name, username })
          });
          const data = await res.json();
          
          if (!res.ok) throw new Error(data.error || 'Failed to request OTP');
          
          // Switch to OTP entering view
          isOtpSentState = true;
          document.getElementById('formFields').classList.add('hidden');
          document.getElementById('otpSection').classList.remove('hidden');
          document.getElementById('otp').setAttribute('required', 'true');
          
          submitBtn.disabled = false;
          submitBtn.innerHTML = `<span>Verify & Sign Up</span><span class="material-symbols-outlined text-[18px]">vpn_key</span>`;
          
          showCustomModal('OTP Sent!', 'Please check your email address for the 6-digit verification code.', 'success');
          startCooldown(60);
        } catch (err) {
          console.error(err);
          showCustomModal('Request Failed', err.message);
          submitBtn.disabled = false;
          submitBtn.innerHTML = `<span>Send OTP</span><span class="material-symbols-outlined text-[18px]">arrow_forward</span>`;
        }
      } else {
        // Step 2: Verify OTP
        const otp = document.getElementById('otp').value.trim();
        if (otp.length !== 6) {
          showCustomModal('Invalid Code', 'Please enter a 6-digit verification code.');
          return;

        const submitBtn = document.getElementById('submitBtn');
        submitBtn.disabled = true;
        submitBtn.innerHTML = `<span>Verifying...</span><span class="material-symbols-outlined text-[18px] animate-spin">sync</span>`;
        
        try {
          const verifyRes = await fetch('/api/auth/verify-otp', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'include',
            body: JSON.stringify({ email, otp })
          });
          const verifyData = await verifyRes.json();
          
          if (!verifyRes.ok) throw new Error(verifyData.error || 'Verification failed');

        }
      }
    }
